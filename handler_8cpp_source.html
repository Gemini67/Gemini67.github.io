420 template< typename T>
421  Handler::ArgResult Handler::processArg( const T& arg,
422  const string& argString,
423  detail::ArgListParser::const_iterator& ai,
424  const detail::ArgListParser::const_iterator& end)
425 {
426 
427  detail::TypedArgBase* hdl = mSubGroupArgs.findArg( arg);
428 
429 
430  if (hdl != nullptr)
431  {
432  handleIdentifiedArg( hdl, argString);
433 
434  Handler* subArgHandler = static_cast< detail::TypedArgSubGroup*>( hdl)->obj();
435  ++ai;
436 
437  // we may only advance the main iterator if the argument is (still)
438  // handled by the sub-argument
439  detail::ArgListParser::const_iterator subAI( ai);
440  while ((subAI != end) && subArgHandler->evalSingleArgument( subAI, end))
441  {
442  ai = subAI;
443  ++subAI;
444  } // end while
445 
446  mpLastArg = nullptr;
447  return arConsumed;
448  } // end if
449 
450  mpLastArg = hdl = mArguments.findArg( arg);
451  if (hdl == nullptr)
452  return arUnknown;
453 
454  // an argument that we know
455  if (hdl->valueMode() == detail::TypedArgBase::vmUnknown)
456  throw runtime_error( "Value mode not set for argument '" + argString + "'");
457 
458  if (hdl->valueMode() == detail::TypedArgBase::vmNone)
459  {
460  // no value needed
461  handleIdentifiedArg( hdl, argString);
462  return arConsumed;
463  } // end if
464 
465  // check if the next element in the list is a value
466  detail::ArgListParser::const_iterator ait2( ai);
467  ait2.remArgStrAsVal();
468  ++ait2;
469 
470  if ((hdl->valueMode() == detail::TypedArgBase::vmRequired) &&
471  ((ait2 == end) || (ait2->mElementType != detail::ArgListElement::etValue)))
472  throw invalid_argument( "Argument '" + argString + "' requires value(s)");
473 
474  if (((ait2 == end) || (ait2->mElementType != detail::ArgListElement::etValue)) &&
475  (hdl->valueMode() == detail::TypedArgBase::vmUnknown))
476  {
477  handleIdentifiedArg( hdl, argString);
478  } else if ((ait2 != end) && (ait2->mElementType == detail::ArgListElement::etValue))
479  {
480  // first process the argument, *then* assign ait2 to ai
481  // reason: arg and argString are references of ai->, so if we change ai
482  // the references are still on the old value(s)
483  handleIdentifiedArg( hdl, argString, ait2->mValue);
484  ai = ait2;
485  } else
486  {
487  handleIdentifiedArg( hdl, argString);
488  } // end if
489 
490  return arConsumed;
491 } // end Handler::processArg

509 Handler::ArgResult
510  Handler::evalSingleArgument( detail::ArgListParser::const_iterator& ai,
511  const detail::ArgListParser::const_iterator& end)
512 {
513 
514  switch (ai->mElementType)
515  {
516  case detail::ArgListElement::etValue:
517  if ((mpLastArg != nullptr) && mpLastArg->takesMultiValue())
518  {
519  mpLastArg->calledAssign( mReadingArgumentFile, ai->mValue);
520  return arConsumed;
521  } else if (detail::TypedArgBase* hdl = mArguments.findArg( '-'))
522  {
523  handleIdentifiedArg( hdl, string( "-"), ai->mValue);
524  return arConsumed;
525  } // end if
526  break;
527 
528  case detail::ArgListElement::etSingleCharArg:
529  return processArg( ai->mArgChar, string( 1, ai->mArgChar), ai, end);
530 
531  case detail::ArgListElement::etStringArg:
532  return processArg( ai->mArgString, ai->mArgString, ai, end);
533 
534  case detail::ArgListElement::etControl:
535  if (ai->mArgChar == '(')
536  {
537  if (mpOpeningBracketHdlr.empty())
538  return arUnknown;
539  mpOpeningBracketHdlr();
540  } else if (ai->mArgChar == ')')
541  {
542  if (mpClosingBracketHdlr.empty())
543  return arUnknown;
544  mpClosingBracketHdlr();
545  } else
546  {
547  if (mpExclamationMarkHdlr.empty())
548  return arUnknown;
549  mpExclamationMarkHdlr();
550  } // end if
551 
552  return arConsumed;
553 
554  default:
555  throw invalid_argument( "Got invalid element in argument list");
556 
557  } // end switch
558 
559  return arUnknown;
560 } // end Handler::evalSingleArgument

822 bool Handler::validArguments( string& constraint_arg_list) const
823 {
824 
825  common::Tokenizer tok( constraint_arg_list, ';');
826  string new_constraint_arg_list;
827 
828 
829  for (auto const& it : tok)
830  {
831  const string::size_type comma_pos = it.find( ',');
832  string current_arg;
833  if (comma_pos == string::npos)
834  {
835  detail::TypedArgBase* arg = nullptr;
836  if (it.length() == 1)
837  {
838  // only a short argument
839  arg = mArguments.findArg( it[ 0]);
840  } else
841  {
842  // only a long argument
843  arg = mArguments.findArg( it);
844  } // end if
845 
846  // found the handler for this argument?
847  if (arg == nullptr)
848  return false;
849 
850  // argument is valid but may use both short and long format
851  current_arg = arg->argSpec();
852  } else
853  {
854  // both short and long argument
855  const string first( it, 0, comma_pos);
856  const string second( it, comma_pos + 1, string::npos);
857  detail::TypedArgBase* arg_short = nullptr;
858  detail::TypedArgBase* arg_long = nullptr;
859 
860  if (first.length() == 1)
861  {
862  arg_short = mArguments.findArg( first[ 0]);
863  arg_long = mArguments.findArg( second);
864  } else
865  {
866  arg_short = mArguments.findArg( second[ 0]);
867  arg_long = mArguments.findArg( first);
868  } // end if
869 
870  // found both the short and long argument?
871  if ((arg_short == nullptr) || (arg_long == nullptr))
872  return false;
873 
874  // do the short and long format (from the constraint string) really
875  // belong to the same handler?
876  if (arg_short != arg_long)
877  throw runtime_error( "argument short/long mismatch");
878 
879  current_arg = arg_short->argSpec();
880  } // end if
881 
882  // argument from the list is valid
883  if (!new_constraint_arg_list.empty())
884  new_constraint_arg_list.append( ";");
885  new_constraint_arg_list.append( current_arg);
886  } // end for
887 
888  // when we get here, all the arguments in the constraint string were valid
889  // maybe one or more argument specification was expanded, take the new string
890  constraint_arg_list = new_constraint_arg_list;
891 
892  return true;
893 } // end Handler::validArguments
