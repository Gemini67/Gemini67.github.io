158 template< typename T, typename TF>
159  RangeStringIterator< T, TF>& RangeStringIterator< T, TF>::operator ++()
160 {
161  if (mPos == std::string::npos)
162  throw std::runtime_error( "attempt to increment after end-of-range");
163 
164  ++(*mpRanger);
165  if (*mpRanger == mpRanger->end())
166  {
167  mPos += mMainExpression.matchedExpression().length();
168  if (mPos >= mSource.length())
169  {
170  // end reached
171  mPos = std::string::npos;
172  mpRanger.reset();
173  mCurrentValue = TF();
174  return *this;
175  } // end if
176 
177  if (mSource[ mPos] != ',')
178  throw std::runtime_error( "invalid character in range string");
179 
180  mPos++;
181  mMainExpression.parseString( mSource.substr( mPos));
182 
183  if (mMainExpression.matchedExpression().length() == 0)
184  throw std::runtime_error( "invalid expression in range string");
185 
186  mpRanger.reset( createRanger( mMainExpression));
187  } // end if
188 
189  mCurrentValue = static_cast< TF>( *mpRanger);

226 template< typename T, typename TF>
227  typename RangeStringIterator< T, TF>::Ranger*
228  RangeStringIterator< T, TF>::createRanger( const RangeExpression& re)
229 {
230  if (!re.hasRangeEnd())
231  // single value
232  return new Ranger( static_cast< TF>( re.startValue()));
233 
234  Ranger* newRanger = nullptr;
235 
236  if (re.hasIncrement())
237  // range with increment
238  newRanger = new Ranger( static_cast< TF>( re.startValue()),
239  static_cast< TF>( re.endValue()),
240  static_cast< TF>( re.incrementValue()));
241  else
242  // range without increment
243  newRanger = new Ranger( static_cast< TF>( re.startValue()),
244  static_cast< TF>( re.endValue()));
245 
246  if (re.hasExcludeExpr())
247  {
248  // an exclude expression can be a full-fledged expression string itself
249  // therefore we need a RangeStringIter (myself :-) to evaluate it
250  // this means that an exclude expression could even include an
251  // exclude expression (which includes the values again )...
252  // since we are parsing a string, the danger of endless recursion is
253  // small and we don't check for it
254  for (RangeStringIterator< T, TF> it( re.excludeExpression()), end;
255  it != end; ++it)
256  {
257  newRanger->excludeValue( static_cast< TF>( it));
258  } // end for
259  } // end if
260 
261  return newRanger;
262 } // end RangeStringIterator< T, TF>::createRanger
