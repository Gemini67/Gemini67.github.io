209  ArgListIterator< T, E>& ArgListIterator< T, E>::operator ++()
210 {

216  *this = ArgListIterator( *mpSource, true);
217  } else if (mNextIsValue ||
218  (mRemainingArgumentStringAsValue && (mArgCharPos > 0)))

222  // or

282 template< typename T, typename E>
283  void ArgListIterator< T, E>::determineNextArg()
284 {
285 
286  if (mpArgV[ mArgIndex][ mArgCharPos] == '-')
287  {
288  // double dash: long argument
289  if (mArgCharPos + 1 == mCurrArgStringLen)
290  {
291  mAcceptDashedValue = true;
292  ++mArgIndex;
293  mArgCharPos = 0;
294  operator ++();
295  return;
296  } // end if
297 
298  // two dashes followed by long name
299  std::string argName( &mpArgV[ mArgIndex][ mArgCharPos + 1]);
300  std::string::size_type equalPos = argName.find_first_of( '=');
301  if (equalPos == std::string::npos)
302  {
303  mCurrElement.setArgString( mArgIndex, argName);
304  ++mArgIndex;
305  mArgCharPos = 0;
306  } else
307  {
308  // long argument with =
309  argName.erase( equalPos);
310  mCurrElement.setArgString( mArgIndex, argName);
311  mArgCharPos += equalPos + 2;
312  mNextIsValue = true;
313  } // end if
314  } else if (mCurrArgStringLen == mArgCharPos + 1)
315  {
316  // one dash -> next is argument character
317  mCurrElement.setArgChar( mArgIndex, mArgCharPos,
318  mpArgV[ mArgIndex][ mArgCharPos]);
319  ++mArgIndex;
320  mArgCharPos = 0;
321  } else
322  {
323  // format: -<char>[<char>]
324  mCurrElement.setArgChar( mArgIndex, mArgCharPos,
325  mpArgV[ mArgIndex][ mArgCharPos]);
326  ++mArgCharPos;
327  if (mArgCharPos >= mCurrArgStringLen)
328  {
329  ++mArgIndex;
330  mArgCharPos = 0;
331  } // end if
332  } // end if
333 
334 } // end ArgListIterator< T, E>::determineNextArg
