511 template< typename T>
512  void TypedArg< std::vector< T>>::assign( const std::string& value)
513 {
514  common::Tokenizer tok( value, mListSep);
515  for (auto it = tok.begin(); it != tok.end(); ++it)
516  {
517  if ((it != tok.begin()) && (mpCardinality.get() != nullptr))
518  mpCardinality->gotValue();
519 
520  const std::string& listVal( *it);
521 
522  check( listVal);
523 
524  if (!mFormats.empty())
525  {
526  std::string valCopy( listVal);
527  format( valCopy);
528  mDestVar.push_back( boost::lexical_cast< T>( valCopy));
529  } else
530  {
531  mDestVar.push_back( boost::lexical_cast< T>( listVal));
532  } // end if
