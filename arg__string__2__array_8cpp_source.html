137 void splitString( StringVec& arguments, const string& argstring)
138 {
139 
140  string currWord;
141  char usedQuoteChar = '-';
142  bool inQuote = false;
143  bool gotBackslash = false;
144 
145 
146  // priority for parsing:
147  // - backslashes: had a backslash before or a new backslash?
148  // - in a quoted string?
149  // - new quote, single or double?
150  // - space
151  // - all the rest
152  for (auto next_char : argstring)
153  {
154  if (gotBackslash)
155  {
156  currWord.append( 1, next_char);
157  gotBackslash = false;
158  } else if (next_char == '\\')
159  {
160  gotBackslash = true;
161  } else if (inQuote)
162  {
163  if (next_char == usedQuoteChar)
164  {
165  usedQuoteChar = '-';
166  inQuote = false;
167  } else
168  {
169  currWord.append( 1, next_char);
170  } // end if
171  } else if ((next_char == '\'') || (next_char == '"'))
172  {
173  inQuote = true;
174  usedQuoteChar = next_char;
175  } else if (next_char == ' ')
176  {
177  arguments.push_back( currWord);
178  currWord.clear();
179  } else
180  {
181  currWord.append( 1, next_char);
182  } // end if
183  } // end for
184 
185  if (currWord.length() > 0)
186  arguments.push_back( currWord);
