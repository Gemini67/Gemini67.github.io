90  if (arg_spec.length() - ignore_leading_dashes == 1)
91  mChar = arg_spec[ ignore_leading_dashes];
92  else
93  mWord = arg_spec.substr( ignore_leading_dashes);
94  } else
95  {

97  throw invalid_argument( "too many commas, only one allowed");

152 bool ArgumentKey::mismatch( const ArgumentKey& other) const
153 {
154
