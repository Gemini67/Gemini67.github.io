93 void TextBlock::formatLine( ostream& os, const string& line)
94 {
95 
96  common::Tokenizer tokWord( line, ' ');
97  string::size_type currLength = mIndentSpaces.length();
98  bool lineStartsWithDash = false;
99 
100 
101  for (auto tiWord : tokWord)
102  {
103  // check if the current word still fits onto the current line,
104  // otherwise start a new line
105  // allow output length == defined length to include the last column
106  if (tiWord == "nn")
107  {
108  os << endl << mIndentSpaces;
109  currLength = mIndentSpaces.length();
110  if (lineStartsWithDash)
111  {
112  // trick: only indent 1 additional character here
113  // because afterwards a single word without dash will be written,
114  // and because currLength != mIndentSpaces.length(), it will be
115  // preceded by a space
116  // so, instead of introducing a new flag or making the check for the
117  // need of space(s) more complicated, we only add one space here
118  os << " ";
119  ++currLength;
120  } // end if
121  } else if (currLength + tiWord.length() + 1 > mLength)
122  {
123  os << endl << mIndentSpaces;
124  if (lineStartsWithDash)
125  os << " ";
126  os << tiWord;

128  if (lineStartsWithDash)
129  currLength += 2;
130  } else

133  if (currLength != mIndentSpaces.length())
