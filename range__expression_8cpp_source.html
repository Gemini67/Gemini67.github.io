86 bool RangeExpression::parse()
87 {
88 
89  // first there must always be a number
90  if (!::isdigit( mRangeString[ 0]))
91  return false;
92 
93  mNextPos = 0;
94  readNumber( mStartValue);
95 
96  mFound = miStartValue;
97 
98  // range?
99  if ((mNextPos < mRangeString.length()) && (mRangeString[ mNextPos] == '-'))
100  {
101  // have a range, need at least one digit now
102  if (!::isdigit( mRangeString[ ++mNextPos]))
103  return false;
104 
105  readNumber( mEndValue);
106  mFound |= miEndValue;
107 
108  // increment and/or exclude-string can only be used together with a range
109 
110  // increment?
111  if ((mNextPos < mRangeString.length()) && (mRangeString[ mNextPos] == '['))
112  {
113  // have an increment, need at least one digit now
114  if (!::isdigit( mRangeString[ ++mNextPos]))
115  return false;
116 
117  readNumber( mIncrementValue);
118 
119  if (mRangeString[ mNextPos] != ']')
120  return false;
121 
122  ++mNextPos;
123  mFound |= miIncrement;
124  } // end if
125 
126  // exclude-range?
127  if ((mNextPos < mRangeString.length()) && (mRangeString[ mNextPos] == '{'))
128  {
129  string::size_type exclude_begin = ++mNextPos;
130  string::size_type inner_exclude = 0;
131 
132  // find the end of the exclude range
133  // since the exclude-range itself can be a range that contains an
134  // exclude range, have to keep track of the { } we find
135  while (mNextPos < mRangeString.length())
136  {
137  if (mRangeString[ mNextPos] == '{')
138  {
139  ++inner_exclude;
140  } else if (mRangeString[ mNextPos] == '}')
141  {
142  if (inner_exclude > 0)
143  {
144  --inner_exclude;
145  } else
146  {
147  break; // while
148  } // end if
149  } // end if
150  ++mNextPos;
151  } // end while
152 
153  if (mRangeString[ mNextPos] != '}')
154  return false;
155 
156  mExcludeValue.assign( mRangeString, exclude_begin, mNextPos - exclude_begin);
157  ++mNextPos;
158  mFound |= miExclude;
159  } // end if
160  } // end if
161 
162  return true;
163 } // end RangeExpression::parse
